category-s

category-s : class-t
category-s = class
  object-t : type-t*

  arrow-t :
    -- object-t object-t
    -> type-t*

  arrow-eqv-t :
    -- a b arrow-t a b arrow-t
    -> type-t*

  identity :
    -- object-t % a
    -> a a arrow-t

  compose :
    -- a b arrow-t
       b c arrow-t
    -> a c arrow-t

  identity-left :
    -- a b arrow-t % f
    -> a identity f compose, f arrow-eqv-t

  identity-right :
    -- a b arrow-t % f
    -> f b identity compose, f arrow-eqv-t

  compose-associative :
    -- a b arrow-t % f
       b c arrow-t % g
       c d arrow-t % h
    -> f g h compose compose
       f g compose h compose arrow-eqv-t

category-s.arrow-inverse-t

category-s.arrow-inverse-t :
  -- a b arrow-t
     b a arrow-t
  -> type-t
category-s.arrow-inverse-t = lambda
  let f g in
  f g compose a identity arrow-eqv-t
  g f compose b identity arrow-eqv-t

category-s.arrow-unique-t

category-s.arrow-unique-t :
  -- a b arrow-t
     -- a b arrow-t -> type-t
  -> type-t
category-s.arrow-unique-t = lambda
  let f theorem in
  f theorem
  -- a b arrow-t % g
     g theorem
  -> f g arrow-eqv-t

category-s.object-product-t

category-s.object-product-t :
  -- object-t % a
     object-t % b
     object-t % p
     p a arrow-t % fst
     p b arrow-t % snd
  -> type-t
category-s.object-product-t = lambda
  let a b p fst snd in
  -- object-t % q
     q a arrow-t % fst-
     q b arrow-t % snd-
  -> q p arrow-t % m
     lambda let m in
       fst-, m fst compose arrow-eqv-t
       snd-, m snd compose arrow-eqv-t
     m swap arrow-unique-t

  @ [start end] : node-t

  -> node node node-path-t

  -- a b edge-t % edge
  -> a b edge-path-t

  -- a b path-t
     a b path-t
  -> type-t

  -> p p refl-path-eqv-t

  -- a b path-t % p
  -> p b node-path-c link-path-c
     p node-right-path-eqv-t

  -- a b path-t % p
     b c path-t % q
     c d path-t % r
